System Overview

This system manages employee parking allocations, EV charging bookings, and parking waitlists.

Database Status

Database Tables:

  • buildingInfo - Buildings and their parking capacity
  • departmentInfo - Departments linked to buildings
  • employeeInfo - Employee records
  • loginInfo - User authentication
  • parkingInfo - Parking spot allocations
  • evBook - EV charging bookings
  • parkingWaitList - Waiting list

Employee SQL Commands

-- Add new employee INSERT INTO employeeInfo (employeeId, firstName, lastName, employeeStatus, departmentId, age) VALUES (31, 'John', 'Doe', 'FT', 1, 30); -- View all employees SELECT e.*, d.departmentName FROM employeeInfo e JOIN departmentInfo d ON e.departmentId = d.departmentId ORDER BY e.employeeId; -- Delete employee DELETE FROM employeeInfo WHERE employeeId = 31; -- Update employee UPDATE employeeInfo SET departmentId = 2 WHERE employeeId = 31;

Parking Management

Parking Status

Parking SQL Commands

-- View available spots SELECT parkingNum, evCharge, fastCharge FROM parkingInfo WHERE employeeId IS NULL ORDER BY parkingNum; -- Assign parking spot UPDATE parkingInfo SET employeeId = 15 WHERE parkingNum = 7; -- Release parking spot UPDATE parkingInfo SET employeeId = NULL WHERE parkingNum = 7; -- View occupied spots SELECT p.parkingNum, e.firstName, e.lastName FROM parkingInfo p JOIN employeeInfo e ON p.employeeId = e.employeeId ORDER BY p.parkingNum;

Report SQL Queries

-- Employee Status Report SELECT employeeStatus, COUNT(*) as count FROM employeeInfo GROUP BY employeeStatus; -- Parking Utilization SELECT COUNT(*) as total_spots, COUNT(employeeId) as occupied, COUNT(*) - COUNT(employeeId) as available FROM parkingInfo; -- Current Waitlist SELECT w.waitListId, e.firstName, e.lastName, w.waitFrom FROM parkingWaitList w JOIN employeeInfo e ON w.employeeId = e.employeeId WHERE w.parkingNum IS NULL ORDER BY w.waitFrom; -- Department Statistics SELECT d.departmentName, COUNT(DISTINCT e.employeeId) as employees, COUNT(DISTINCT p.parkingNum) as parking_spots FROM departmentInfo d LEFT JOIN employeeInfo e ON d.departmentId = e.departmentId LEFT JOIN parkingInfo p ON e.employeeId = p.employeeId GROUP BY d.departmentName;
